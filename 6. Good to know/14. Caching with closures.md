```js
function cache(value) {
  let _value = value;
  return {
    getValue() {
      return _value;
    },
    setValue(value) {
      _value = value;
    }
  }
}

const cached = cache('2020 facts');

// some time later... 
cached.getValue(); // "2020 facts"
```
