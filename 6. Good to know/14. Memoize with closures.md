```js
function memoize(value) {
  let _value = value;
  return {
    getValue() {
      return _value;
    },
    setValue(value) {
      _value = value;
    }
  }
}

const memo = memoize('2020 facts');

// some time later... 
memo.getValue(); // "2020 facts"
```
