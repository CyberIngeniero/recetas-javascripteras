# Replace switch with object

```js
const animal = "dog";

switch (animal) {
  case "dog":
    console.log("Si si si, simba! 🎺");
    break;
  case "cat":
    console.log("Me pareció ver un lindo gatito");
    break;
}
// 'Si si si, simba! 🎺'

const logAnimal = {
  dog: () => console.log("Si si si, simba! 🎺"),
  cat: () => console.log("Me pareció ver un lindo gatito"),
};

logAnimal[animal]();
// 'Si si si, simba! 🎺'
```
